    Responses in PyAgentic

    PyAgentic provides a sophisticated response system that captures and structures the outputs from agents, tools, and their interactions. The response system is built on Pydantic models, providing type safety, validation, and easy integration with web frameworks like FastAPI.

    Why Structured Responses?

    PyAgentic's response system offers several key advantages:

    • Type Safety: All responses are strongly typed with Pydantic models
    • Structured Data: Capture not just final outputs but also tool execution details
    • API Integration: Perfect for building REST APIs and web services
    • Debugging: Detailed information about tool calls and execution flow
    • Composability: Support for complex multi-agent response aggregation

    Response Architecture

    PyAgentic uses a three-tier response architecture:

    1. ToolResponse: Captures individual tool execution details
    2. AgentResponse: Captures the complete agent execution, including all tool calls
    3. Linked Agent Responses: Nested responses from linked agents

    from pyagentic import Agent, tool, ParamInfo
from typing import List

class CalculatorAgent(Agent):
    """Agent with mathematical tools"""

    __system_message__ = "You are a calculator that performs mathematical operations"

    calculation_history: List[str] = ContextItem(default_factory=list)

    @tool("Add two numbers together")
    def add(
        self, 
        a: float = ParamInfo(required=True, description="First number"),
        b: float = ParamInfo(required=True, description="Second number")
    ) -> str:
        """Add two numbers"""
        result = a + b
        self.context.calculation_history.append(f"{a} + {b} = {result}")
        return f"The sum of {a} and {b} is {result}"

    @tool("Calculate the factorial of a number")
    def factorial(
        self,
        n: int = ParamInfo(required=True, description="Number to calculate factorial for")
    ) -> str:
        """Calculate factorial"""
        if n < 0:
            return "Factorial is not defined for negative numbers"

        result = 1
        for i in range(1, n + 1):
            result *= i

        return f"The factorial of {n} is {result}"

# When you run this agent
agent = CalculatorAgent(model="gpt-4o", api_key="your-api-key")
response = await agent.run("Please add 15 and 27, then calculate the factorial of 5")

# Each tool execution creates a ToolResponse
for tool_response in response.tool_responses:
    print(f"Tool: {tool_response.__class__.__name__}")
    print(f"Output: {tool_response.output}")
    print(f"Raw kwargs: {tool_response.raw_kwargs}")
    print(f"Call depth: {tool_response.call_depth}")

    # Access the actual parameters used (type-safe)
    if hasattr(tool_response, 'a'):  # This was an 'add' call
        print(f"Parameter 'a': {tool_response.a}")
        print(f"Parameter 'b': {tool_response.b}")
    elif hasattr(tool_response, 'n'):  # This was a 'factorial' call
        print(f"Parameter 'n': {tool_response.n}")

    ToolResponse Structure

    Each ToolResponse contains:

    • output: The string returned by the tool function
    • raw_kwargs: The original JSON arguments from OpenAI
    • call_depth: Nesting level for recursive tool calls
    • Tool Parameters: Each parameter becomes a typed field in the response

    class DataProcessor(Agent):
    __system_message__ = "I process data with various tools"

    @tool("Process a list of numbers")
    def process_numbers(
        self,
        numbers: List[float] = ParamInfo(required=True),
        operation: str = ParamInfo(required=True, values=["sum", "average", "max", "min"])
    ) -> str:
        """Process a list of numbers"""
        if operation == "sum":
            result = sum(numbers)
        elif operation == "average":
            result = sum(numbers) / len(numbers)
        elif operation == "max":
            result = max(numbers)
        elif operation == "min":
            result = min(numbers)

        return f"The {operation} of {numbers} is {result}"

    @tool("Format text with styling")
    def format_text(
        self,
        text: str = ParamInfo(required=True),
        style: str = ParamInfo(default="plain", values=["plain", "bold", "italic", "uppercase"])
    ) -> str:
        """Format text with different styles"""
        if style == "bold":
            return f"**{text}**"
        elif style == "italic":
            return f"*{text}*"
        elif style == "uppercase":
            return text.upper()
        return text

# Usage
processor = DataProcessor(model="gpt-4o", api_key="your-api-key")
response = await processor.run("Process the numbers [1, 2, 3, 4, 5] to find their average, then format the result in bold")

# Each tool gets its own response type
for tool_response in response.tool_responses:
    if hasattr(tool_response, 'numbers'):
        # This is a ToolResponse[process_numbers]
        print(f"Numbers processed: {tool_response.numbers}")
        print(f"Operation: {tool_response.operation}")
    elif hasattr(tool_response, 'text'):
        # This is a ToolResponse[format_text] 
        print(f"Text formatted: {tool_response.text}")
        print(f"Style applied: {tool_response.style}")

    class ComprehensiveAgent(Agent):
    """Agent that demonstrates comprehensive response structure"""

    __system_message__ = """
    You are a comprehensive assistant that can perform calculations,
    format text, and maintain conversation history.
    """

    conversation_history: List[str] = ContextItem(default_factory=list)

    @tool("Perform mathematical calculations")
    def calculate(self, expression: str) -> str:
        """Safely evaluate mathematical expressions"""
        try:
            # Simple evaluation (use a safer evaluator in production)
            result = eval(expression.replace('^', '**'))
            return f"Result: {expression} = {result}"
        except Exception as e:
            return f"Error: {str(e)}"

    @tool("Record conversation context")
    def record_context(self, context: str) -> str:
        """Keep track of important conversation points"""
        self.context.conversation_history.append(context)
        return f"Recorded: {context}"

# When you run the agent
agent = ComprehensiveAgent(model="gpt-4o", api_key="your-api-key")
response = await agent.run("Calculate 2 * 3 + 4, and remember that we discussed mathematics")

# AgentResponse structure
print("=== Agent Response Structure ===")
print(f"Final Output: {response.final_output}")
print(f"Number of tool calls: {len(response.tool_responses) if response.tool_responses else 0}")

# Access tool responses
if response.tool_responses:
    for i, tool_response in enumerate(response.tool_responses):
        print(f"\nTool Call {i + 1}:")
        print(f"  Type: {tool_response.__class__.__name__}")
        print(f"  Output: {tool_response.output}")
        print(f"  Parameters used:")

        # Dynamically access parameters based on tool type
        for field_name, field_info in tool_response.model_fields.items():
            if field_name not in ['raw_kwargs', 'call_depth', 'output']:
                value = getattr(tool_response, field_name)
                print(f"    {field_name}: {value}")

    AgentResponse Structure

    Each AgentResponse contains:

    • final_output: The agent's final natural language response
    • tool_responses: List of all ToolResponse objects (if tools were used)
    • agent_responses: List of responses from linked agents (if any were called)

    class ResearchAgent(Agent):
    """Agent that performs multi-step research tasks"""

    __system_message__ = """
    You are a research assistant that can search for information,
    analyze data, and generate reports. Use multiple tools as needed.
    """

    research_data: List[dict] = ContextItem(default_factory=list)
    analysis_results: List[str] = ContextItem(default_factory=list)

    @tool("Search for information on a topic")
    def search_information(self, topic: str, sources: int = 3) -> str:
        """Search for information (simulated)"""
        # Simulate search results
        results = [f"Result {i+1} for '{topic}'" for i in range(sources)]

        research_entry = {
            "topic": topic,
            "sources": sources,
            "results": results,
            "timestamp": "2024-01-01"
        }
        self.context.research_data.append(research_entry)

        return f"Found {sources} sources for '{topic}': {', '.join(results)}"

    @tool("Analyze collected research data")
    def analyze_data(self, focus_area: str) -> str:
        """Analyze the collected research data"""
        if not self.context.research_data:
            return "No research data available for analysis"

        analysis = f"Analysis of {focus_area}: Found {len(self.context.research_data)} research entries"
        analysis += f" with {sum(len(entry['results']) for entry in self.context.research_data)} total sources"

        self.context.analysis_results.append(analysis)
        return analysis

    @tool("Generate summary report")
    def generate_report(self, report_type: str = "comprehensive") -> str:
        """Generate a summary report of research and analysis"""
        if not self.context.analysis_results:
            return "No analysis available for report generation"

        report = f"=== {report_type.upper()} RESEARCH REPORT ===\n"
        report += f"Data Sources: {len(self.context.research_data)} research queries\n"
        report += f"Analyses Performed: {len(self.context.analysis_results)}\n"
        report += "\nKey Findings:\n"

        for i, analysis in enumerate(self.context.analysis_results, 1):
            report += f"{i}. {analysis}\n"

        return report

# Multi-step execution
research_agent = ResearchAgent(model="gpt-4o", api_key="your-api-key")
response = await research_agent.run("""
Please research artificial intelligence and machine learning, 
analyze the data focusing on recent trends, and generate a comprehensive report.
""")

print(f"Final Response: {response.final_output}")
print(f"\nTools executed: {len(response.tool_responses)}")

# Detailed tool execution trace
for i, tool_response in enumerate(response.tool_responses):
    print(f"\n--- Tool Execution {i+1} ---")
    print(f"Tool: {tool_response.__class__.__name__}")
    print(f"Output: {tool_response.output[:100]}...")  # Truncate for display

    # Show the execution flow
    if hasattr(tool_response, 'topic'):
        print(f"→ Searched for: {tool_response.topic}")
    elif hasattr(tool_response, 'focus_area'):
        print(f"→ Analyzed: {tool_response.focus_area}")
    elif hasattr(tool_response, 'report_type'):
        print(f"→ Generated: {tool_response.report_type} report")

    class SpecialistAgent(Agent):
    """A specialist agent that performs specific tasks"""

    __system_message__ = "I am a specialist that performs focused analysis"
    __description__ = "Perform specialized analysis tasks"

    analyses: List[str] = ContextItem(default_factory=list)

    @tool("Perform specialized analysis")
    def analyze(self, data: str, method: str) -> str:
        """Perform analysis using specified method"""
        analysis_result = f"Analysis of '{data}' using {method}: Found 3 key patterns"
        self.context.analyses.append(analysis_result)
        return analysis_result

class CoordinatorAgent(Agent):
    """Agent that coordinates with specialists"""

    __system_message__ = """
    I coordinate complex tasks by working with specialist agents.
    I can delegate specialized work while managing the overall process.
    """

    # Link the specialist
    specialist: SpecialistAgent

    task_queue: List[str] = ContextItem(default_factory=list)

    @tool("Add task to coordination queue")
    def queue_task(self, task_description: str) -> str:
        """Add a task to the coordination queue"""
        self.context.task_queue.append(task_description)
        return f"Queued task: {task_description}"

# Create the linked system
specialist = SpecialistAgent(model="gpt-4o", api_key="your-api-key")
coordinator = CoordinatorAgent(
    model="gpt-4o", 
    api_key="your-api-key",
    specialist=specialist
)

# Execute with linked agents
response = await coordinator.run("""
Please analyze the sales data from Q3 using statistical methods,
and add a follow-up task to review the results next week.
""")

print(f"Main Response: {response.final_output}")

# Check tool responses (coordinator's direct tools)
if response.tool_responses:
    print(f"\nDirect tool calls: {len(response.tool_responses)}")
    for tool_response in response.tool_responses:
        print(f"  - {tool_response.__class__.__name__}: {tool_response.output}")

# Check linked agent responses
if response.agent_responses:
    print(f"\nLinked agent calls: {len(response.agent_responses)}")
    for agent_response in response.agent_responses:
        print(f"  - Final output: {agent_response.final_output}")

        # Linked agents can also have tool responses
        if hasattr(agent_response, 'tool_responses') and agent_response.tool_responses:
            print(f"    Tools used by linked agent: {len(agent_response.tool_responses)}")
            for tool_response in agent_response.tool_responses:
                print(f"      → {tool_response.__class__.__name__}: {tool_response.output}")

    def analyze_agent_execution(response):
    """Detailed analysis of agent execution"""

    print("=== EXECUTION ANALYSIS ===")
    print(f"Final Output Length: {len(response.final_output)} characters")

    # Tool execution analysis
    if hasattr(response, 'tool_responses') and response.tool_responses:
        print(f"\nTool Executions: {len(response.tool_responses)}")

        tool_types = {}
        for tool_response in response.tool_responses:
            tool_name = tool_response.__class__.__name__
            tool_types[tool_name] = tool_types.get(tool_name, 0) + 1

        for tool_name, count in tool_types.items():
            print(f"  - {tool_name}: {count} times")

        # Parameter analysis
        print("\nParameter Usage:")
        for i, tool_response in enumerate(response.tool_responses):
            print(f"  Tool {i+1}: {tool_response.__class__.__name__}")
            for field_name, field_info in tool_response.model_fields.items():
                if field_name not in ['raw_kwargs', 'call_depth', 'output']:
                    value = getattr(tool_response, field_name)
                    print(f"    {field_name}: {type(value).__name__} = {value}")

    # Linked agent analysis
    if hasattr(response, 'agent_responses') and response.agent_responses:
        print(f"\nLinked Agent Calls: {len(response.agent_responses)}")

        for i, agent_response in enumerate(response.agent_responses):
            print(f"  Agent {i+1}:")
            print(f"    Output: {len(agent_response.final_output)} chars")

            if hasattr(agent_response, 'tool_responses') and agent_response.tool_responses:
                print(f"    Nested tools: {len(agent_response.tool_responses)}")

from pydantic import BaseModel
import json

app = FastAPI()

class TaskRequest(BaseModel):
    task: str
    priority: str = "normal"

@app.post("/execute-task")
async def execute_task(request: TaskRequest):
    """Execute a task using the agent"""

    # Create and run agent
    agent = MyAgent(model="gpt-4o", api_key="your-api-key")
    response = await agent.run(request.task)

    # The response is already a Pydantic model, so it can be returned directly
    return response

@app.get("/task-analysis/{task_id}")
async def analyze_task_execution(task_id: str):
    """Analyze the execution of a previous task"""

    # Load response from database (as JSON)
    response_json = load_response_from_db(task_id)

    # Reconstruct the response object
    response = MyAgentResponse(**response_json)

    # Perform analysis
    analysis = {
        "final_output_length": len(response.final_output),
        "tools_used": len(response.tool_responses) if response.tool_responses else 0,
        "agents_called": len(response.agent_responses) if response.agent_responses else 0,
    }

    if response.tool_responses:
        analysis["tool_details"] = [
            {
                "type": tool_response.__class__.__name__,
                "output_length": len(tool_response.output),
                "parameters": {
                    field: getattr(tool_response, field)
                    for field in tool_response.model_fields
                    if field not in ['raw_kwargs', 'call_depth', 'output']
                }
            }
            for tool_response in response.tool_responses
        ]

    return analysis

def save_response_to_database(response, task_id: str):

    def safe_response_processing(response):
    """Safely process responses that may not have tools or linked agents"""

    # Always available
    print(f"Final response: {response.final_output}")

    # Check for tool responses
    if hasattr(response, 'tool_responses') and response.tool_responses:

        for agent_response in response.agent_responses:
            print(f"  - Agent output: {agent_response.final_output}")

        print("No linked agents were called")

    from pydantic import ValidationError

def validate_and_process_response(response_data: dict):
    """Validate response data before processing"""

    try:
        # Attempt to create response object
        response = MyAgentResponse(**response_data)

        # Process validated response
        return process_response(response)

    except ValidationError as e:
        print(f"Response validation failed: {e}")
        return {"error": "Invalid response structure"}

def robust_tool_response_handling(response):
    """Handle tool responses robustly"""

    if not hasattr(response, 'tool_responses') or not response.tool_responses:
        return "No tool executions to process"

    results = []
    for tool_response in response.tool_responses:
        try:
            # Process each tool response
            tool_info = {
                "type": tool_response.__class__.__name__,
                "success": True,
                "output": tool_response.output,
                "parameters": {}
            }

            # Safely extract parameters
            for field_name, field_info in tool_response.model_fields.items():
                if field_name not in ['raw_kwargs', 'call_depth', 'output']:
                    try:
                        tool_info["parameters"][field_name] = getattr(tool_response, field_name)
                    except AttributeError:
                        tool_info["parameters"][field_name] = "N/A"

            results.append(tool_info)

        except Exception as e:
            results.append({
                "type": "Unknown",
                "success": False,
                "error": str(e)
            })

    return results

import hashlib
import json

class ResponseCache:
    """Cache for agent responses to improve performance"""

    def __init__(self):
        self._cache = {}

    def cache_key(self, input_text: str, agent_config: dict) -> str:
        """Generate cache key for input and agent configuration"""
        cache_data = {
            "input": input_text,
            "config": agent_config
        }
        return hashlib.md5(json.dumps(cache_data, sort_keys=True).encode()).hexdigest()

    def get_response(self, input_text: str, agent_config: dict):
        """Get cached response if available"""
        key = self.cache_key(input_text, agent_config)
        return self._cache.get(key)

    def cache_response(self, input_text: str, agent_config: dict, response):
        """Cache a response"""
        key = self.cache_key(input_text, agent_config)
        self._cache[key] = response.model_dump()  # Store as dict

    def load_response(self, input_text: str, agent_config: dict, response_class):
        """Load and reconstruct cached response"""
        cached_data = self.get_response(input_text, agent_config)
        if cached_data:
            return response_class(**cached_data)
        return None

    from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from typing import Optional, List

app = FastAPI(title="PyAgentic Agent API")

class AgentRequest(BaseModel):
    input: str
    agent_type: str = "general"
    parameters: Optional[dict] = None

class AgentExecutionResponse(BaseModel):
    success: bool
    final_output: str
    tools_used: int
    execution_time: float
    tool_details: Optional[List[dict]] = None
    agent_calls: Optional[List[dict]] = None

@app.post("/execute", response_model=AgentExecutionResponse)
async def execute_agent(request: AgentRequest):
    """Execute an agent and return structured response"""

    import time
    start_time = time.time()

    try:
        # Create appropriate agent based on type
        agent = create_agent(request.agent_type, request.parameters)

        # Execute agent
        response = await agent.run(request.input)

        # Structure the response for API
        api_response = AgentExecutionResponse(
            success=True,
            final_output=response.final_output,
            tools_used=len(response.tool_responses) if response.tool_responses else 0,
            execution_time=time.time() - start_time
        )

        # Add tool details if available
        if hasattr(response, 'tool_responses') and response.tool_responses:
            api_response.tool_details = [
                {
                    "tool_type": tool_response.__class__.__name__,
                    "output_length": len(tool_response.output),
                    "parameters": {
                        field: getattr(tool_response, field)
                        for field in tool_response.model_fields
                        if field not in ['raw_kwargs', 'call_depth', 'output']
                    }
                }
                for tool_response in response.tool_responses
            ]

        # Add agent call details if available
        if hasattr(response, 'agent_responses') and response.agent_responses:
            api_response.agent_calls = [
                {
                    "output_length": len(agent_response.final_output),
                    "nested_tools": len(agent_response.tool_responses) if hasattr(agent_response, 'tool_responses') and agent_response.tool_responses else 0
                }
                for agent_response in response.agent_responses
            ]

        return api_response

    except Exception as e:
        raise HTTPException(status_code=500, detail=f"Agent execution failed: {str(e)}")
