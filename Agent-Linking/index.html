    Agent Linking in PyAgentic

    Agent linking is a powerful feature in PyAgentic that enables agents to call and coordinate with other agents as if they were tools. This creates sophisticated multi-agent systems where specialized agents can collaborate to solve complex problems while maintaining clean separation of concerns.

    Why Use Agent Linking?

    Agent linking provides several key benefits:

    • Specialization: Create focused agents for specific domains and link them together
    • Modularity: Build complex systems from smaller, reusable agent components
    • Scalability: Distribute workload across multiple specialized agents
    • Maintainability: Keep each agent focused on a single responsibility
    • Composability: Combine agents in different ways for different use cases

    How Agent Linking Works

    When you declare an agent type as a class attribute with a type annotation, PyAgentic automatically:

    1. Detects the linked agent using type analysis during metaclass processing
    2. Registers it as a callable tool that the main agent can invoke
    3. Handles the communication between agents transparently
    4. Manages response aggregation from both tools and linked agents

class MathAgent(Agent):
    """Specialized agent for mathematical calculations"""

    __system_message__ = """
    You are a mathematical calculation specialist. You perform precise calculations
    and provide detailed explanations of mathematical operations.
    """

    __description__ = "Perform mathematical calculations and analysis"

    calculation_history: List[str] = ContextItem(default_factory=list)

    @tool("Perform basic arithmetic operations")
    def calculate(self, expression: str) -> str:
        """Safely evaluate mathematical expressions"""
        try:
            # Simple evaluation (in practice, use a safer evaluator)
            result = eval(expression.replace('^', '**'))
            self.context.calculation_history.append(f"{expression} = {result}")
            return f"Result: {expression} = {result}"
        except Exception as e:
            return f"Error calculating {expression}: {str(e)}"

    @tool("Get calculation history")
    def get_history(self) -> str:
        """Retrieve previous calculations"""
        if not self.context.calculation_history:
            return "No calculations performed yet"
        return "Calculation History:\n" + "\n".join(self.context.calculation_history)

class WritingAgent(Agent):
    """Specialized agent for writing and text analysis"""

    __system_message__ = """
    You are a writing and text analysis specialist. You help with writing,
    editing, summarizing, and analyzing text content.
    """

    __description__ = "Analyze and improve text content"

    documents: List[str] = ContextItem(default_factory=list)

    @tool("Analyze text for readability and style")
    def analyze_text(self, text: str) -> str:
        """Analyze text content"""
        word_count = len(text.split())
        sentence_count = text.count('.') + text.count('!') + text.count('?')
        avg_sentence_length = word_count / max(sentence_count, 1)

        analysis = f"""
        Text Analysis:
        - Word count: {word_count}
        - Sentence count: {sentence_count}
        - Average sentence length: {avg_sentence_length:.1f} words
        - Readability: {'Good' if avg_sentence_length < 20 else 'Could be improved'}
        """

        self.context.documents.append(text)
        return analysis.strip()

    @tool("Summarize text content")
    def summarize(self, text: str, max_sentences: int = 3) -> str:
        """Create a summary of the text"""
        sentences = text.replace('!', '.').replace('?', '.').split('.')
        sentences = [s.strip() for s in sentences if s.strip()]

        # Simple summarization (take first few sentences)
        summary_sentences = sentences[:max_sentences]
        summary = '. '.join(summary_sentences) + '.'

        return f"Summary: {summary}"

class CoordinatorAgent(Agent):
    """Main agent that coordinates between math and writing specialists"""

    __system_message__ = """
    You are an intelligent coordinator that helps users with both mathematical
    and writing tasks. You have access to specialized agents for:

    - Mathematical calculations and analysis (math_assistant)
    - Writing analysis and text processing (writing_assistant)

    Determine which specialist agent to call based on the user's request,
    and coordinate between them when needed.
    """

    # Link the specialized agents
    math_assistant: MathAgent
    writing_assistant: WritingAgent

    task_log: List[str] = ContextItem(default_factory=list)

    @tool("Log a completed task")
    def log_task(self, description: str) -> str:
        """Keep track of completed tasks"""
        self.context.task_log.append(description)
        return f"Logged task: {description}"

# Usage example
math_agent = MathAgent(model="gpt-4o", api_key="your-api-key")
writing_agent = WritingAgent(model="gpt-4o", api_key="your-api-key")

coordinator = CoordinatorAgent(
    model="gpt-4o",
    api_key="your-api-key",
    math_assistant=math_agent,      # Pass the agent instances
    writing_assistant=writing_agent
)

# The coordinator can now call either specialist
response = await coordinator.run("""
Can you calculate the average of these numbers: 15, 23, 31, 42, 18
And then analyze the readability of this explanation I wrote about it?

'The average of a set of numbers is calculated by adding all numbers together 
and dividing by the count of numbers. This gives us the central tendency of the data.'
""")

    class TranslationAgent(Agent):

    __description__ = "Translate text between languages"

    translation_history: List[dict] = ContextItem(default_factory=list)

class SentimentAgent(Agent):
    """Agent that analyzes sentiment in text"""

    __system_message__ = "You analyze the emotional sentiment of text content"
    __description__ = "Analyze sentiment and emotional tone"

    # This agent can also use the translation agent
    translator: TranslationAgent

    sentiment_analyses: List[dict] = ContextItem(default_factory=list)

    @tool("Analyze sentiment of text")
    def analyze_sentiment(self, text: str, language: str = "english") -> str:
        """Analyze the sentiment of text"""
        # Simplified sentiment analysis
        positive_words = ["good", "great", "excellent", "happy", "love"]
        negative_words = ["bad", "terrible", "awful", "sad", "hate"]

        text_lower = text.lower()
        positive_count = sum(word in text_lower for word in positive_words)
        negative_count = sum(word in text_lower for word in negative_words)

        if positive_count > negative_count:
            sentiment = "Positive"
        elif negative_count > positive_count:
            sentiment = "Negative"
        else:
            sentiment = "Neutral"

        analysis = {
            "text": text,
            "sentiment": sentiment,
            "confidence": abs(positive_count - negative_count) / max(len(text.split()), 1),
            "language": language
        }

        self.context.sentiment_analyses.append(analysis)
        return f"Sentiment: {sentiment} (confidence: {analysis['confidence']:.2f})"

class MultilingualContentAgent(Agent):
    """Agent that processes multilingual content using both translation and sentiment"""

    __system_message__ = """
    You process multilingual content by coordinating translation and sentiment analysis.
    You can translate content and then analyze sentiment, or analyze sentiment and 
    then translate, depending on the user's needs.
    """

    # Link to both peer agents
    translator: TranslationAgent
    sentiment_analyzer: SentimentAgent

    processed_content: List[dict] = ContextItem(default_factory=list)

    @tool("Process multilingual content")
    def process_content(self, content: str, source_language: str, target_language: str) -> str:
        """Process content through translation and sentiment analysis"""

        # This agent orchestrates the other two agents
        result = {
            "original_content": content,
            "source_language": source_language,
            "target_language": target_language,
            "processed": True
        }

        self.context.processed_content.append(result)

        return f"""
        Content processed:
        - Original language: {source_language}
        - Target language: {target_language}
        - Content length: {len(content)} characters

        Use translator and sentiment_analyzer tools for detailed processing.
        """

    reporter: ReportAgent

# The response object contains:
# - response.final_output: The main agent's final response
# - response.tool_responses: List of direct tool responses (if any)
# - response.agent_responses: List of responses from linked agents
print(f"Final output: {response.final_output}")
for agent_response in response.agent_responses:
    print(f"Agent response: {agent_response.final_output}")

# Proper initialization respecting dependencies
def create_application_stack(model: str, api_key: str) -> ApplicationAgent:
    """Factory function that ensures proper initialization order"""

    # 1. Create leaf agents first (no dependencies)
    config_agent = ConfigurationAgent(model=model, api_key=api_key)

    # 2. Create agents with dependencies
    db_agent = DatabaseAgent(
        model=model, 
        api_key=api_key,
        config=config_agent  # Provide dependency
    )

    # 3. Create root coordinator
    app_agent = ApplicationAgent(
        model=model,
        api_key=api_key,
        config=config_agent,
        database=db_agent
    )

    return app_agent

    class RouterAgent(Agent):
    """Agent that routes requests to appropriate specialists"""

    __system_message__ = """
    I am a smart router that selects the appropriate specialist agent based on
    the user's request. I analyze the request type and route to:
    - text_specialist: For text processing and analysis
    - math_specialist: For mathematical calculations
    - data_specialist: For data analysis and processing
    """

    text_specialist: TextAgent
    math_specialist: MathAgent  
    data_specialist: DataAgent

    routing_history: List[dict] = ContextItem(default_factory=list)

    @tool("Route request to appropriate specialist")
    def route_request(self, request_type: str, details: str) -> str:
        """Route request based on type analysis"""

        routing_decision = {
            "request_type": request_type,
            "details": details,
            "timestamp": "2024-01-01",
            "routed_to": f"{request_type}_specialist"
        }

        self.context.routing_history.append(routing_decision)

        return f"Routing {request_type} request to {request_type}_specialist"

By following these patterns and best practices, you can build scalable agent networks that maintain clean separation of concerns while providing powerful collaborative capabilities.
